void selectionsort(int x[],int n)
{ int i,j,min;
    for(i=0;i<n;i++)
    {
        for(j=i+1;j<n;j++)
        {
          if(x[min] > x[j])  
          min = j;
        }
        if(min!=i)
        swap(&x[i],&x[min]);
    }
}
void swap(int *a,int *b) 
{
    int *t;
    *t = *a;
    *a = *b;
    *b = *t;
}
