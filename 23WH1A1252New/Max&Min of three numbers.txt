#include<stdio.h>
void main()
{
    int a,b,c;
    printf("Enter three values: ");
    scanf("%d %d %d",&a,&b,&c);
    if (a>b && a>c)
    {
        printf("\n%d is greatest",a);
    }
    else if (b>a && b>c)
    {
        printf("\n%d is greatest",b);
    }
    else
    {
        printf("\n%d is greatest",c);
    }
    if (a<b && a<c)
    {
        printf("\n%d is least",a);
    }
    else if (b<a && b<c)
    {
        printf("\n%d is least",b);
    }
    else
    {
        printf("\n%d is least",c);
    }
}
